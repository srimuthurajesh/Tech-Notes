<project>
  <modelVersion>4.0.0</modelVersion>
  <!-- Project coordinates -->
  <groupId>com.myApp</groupId>
  <artifactId>myApp</artifactId>
  <version>1.0Final</version>
  <packaging>jar</packaging>
  <name>myApp</name>
  
  <dependenices>
    <dependency>
      <groupId>jUnit</groupId>
      <artifactId>jUnit</artifactId>
      <version>3.8.1</version>
      <scope>jUnit</scope>
    </dependency>
  </dependenices>
</project>
